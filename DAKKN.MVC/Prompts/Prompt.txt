BEGIN TRANSACTION;

-- 1. Declare User and Role ID variables
DECLARE @UserId UNIQUEIDENTIFIER = NEWID();
DECLARE @AdminRoleId UNIQUEIDENTIFIER;

-- 2. Find the Admin Role ID from AspNetRoles table
SELECT @AdminRoleId = Id FROM AspNetRoles WHERE Name = 'Admin';

-- If the Admin role doesn't exist yet, create it
IF @AdminRoleId IS NULL
BEGIN
    SET @AdminRoleId = NEWID();
    INSERT INTO AspNetRoles (Id, Name, NormalizedName, ConcurrencyStamp)
    VALUES (@AdminRoleId, 'Admin', 'ADMIN', NEWID());
END

-- 3. Insert the User (System Admin)
INSERT INTO AspNetUsers (
    Id, 
    UserName, 
    NormalizedUserName, 
    Email, 
    NormalizedEmail, 
    EmailConfirmed, 
    PasswordHash, 
    SecurityStamp, 
    ConcurrencyStamp, 
    PhoneNumberConfirmed, 
    TwoFactorEnabled, 
    LockoutEnabled, 
    AccessFailedCount,
    CreatedAt,
    IsDeleted,
    IsActive,
    CreatedBy,
    FullName,
    BirthDate,
    Gender,
    Language
)
VALUES (
    @UserId,
    'dev.mohamed104saber@gmail.com',
    'DEV.MOHAMED104SABER@GMAIL.COM',
    'dev.mohamed104saber@gmail.com',
    'DEV.MOHAMED104SABER@GMAIL.COM',
    1, -- EmailConfirmed (true)
    'AQAAAAIAAYagAAAAEAQt1xhX+om8CqGcY2YHTTH1wVPSLXLkQUrq32nMBZ4mpuXrfrVOT+LhvBkWHHzD8g==', -- Valid password hash for "Admin@123"
    NEWID(), -- SecurityStamp
    NEWID(), -- ConcurrencyStamp
    1, -- PhoneNumberConfirmed (true)
    0, -- TwoFactorEnabled (false)
    1, -- LockoutEnabled (true)
    0, -- AccessFailedCount
    GETUTCDATE(), -- CreatedAt
    0, -- IsDeleted (false)
    1, -- IsActive (true)
    'System', -- CreatedBy
    'System Admin', -- FullName
    '1990-01-01', -- BirthDate
    1, -- Gender (Male = 1, from Gender.cs)
    2 -- Language (en = 2, from LanguageCode.cs)
);

-- 4. Assign the Admin Role to the User
INSERT INTO AspNetUserRoles (UserId, RoleId)
VALUES (@UserId, @AdminRoleId);

COMMIT TRANSACTION;
